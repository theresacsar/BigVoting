057        /**
058         * Compute the Schwartzset of the given graph locally. 
059         * The number of SCCs in the graph and the size of the Schwartzset are written to the logFile.
060         * Returns the list of vertices contained in the Schwartzset.
061         * 
062         * @param graph Contains a directed graph (DiGraph).
063         * @param logFile PrintWriter object of log File.
064         * @return schwartzset List of vertices contained in the schwartzset of graph.
065         */
066        public static List<Integer> SchwartzSet(Digraph graph, PrintWriter logFile){
067                
068                //Reverse the Graph: 
069                 Digraph graphrev = graph.reverse();
070                 
071                 //Compute all SCCs: 
072                 TarjanSCC sccs= new TarjanSCC(graph);
073                 
074                 //Compute the Transitive Closure of the Graph:
075                 TransitiveClosure closure = new TransitiveClosure(graph);
076                 
077                 //schwartzlist contains all SCCs contained in the schwartzset
078                 List<Integer> schwartzlist = new ArrayList<Integer>(); 
079                 
080                 //checked is the list of all SCCs that have already been checked if they are undominated
081                 List<Integer> checked = new ArrayList<Integer>();
082                 
083                 //schwartzset contains the list of vertices contained in the schwartzset
084                 List<Integer> schwartzset = new ArrayList<Integer>();
085                 
086                 //Number of Vertices in the Graph:
087                 int n=graph.V();                
088                 
089                 //other variables: 
090                 int scc; 
091                 boolean schwartz;
092                 
093                 logFile.write("Total SCCs:"+sccs.count()+"\n");
094
095                 /*
096                  * First get all IDs of the SCCs that are undominated.
097                  * The list checked contains all IDs of SCCs that have already been checked. 
098                  * 
099                  */
100                 for(int i=0; i<n; i++){
101                         scc=sccs.id(i);
102                         if(checked.contains(scc)){
103                                 /* 
104                                  * this SCC has already been checked - go to the next
105                                  */
106                         }else{
107                                 /*
108                                  * check if SCC is undominated
109                                  */
110                                 schwartz=true;
111                                 for(int j : graphrev.adj(i)){
112                                         if(sccs.id(j)!=scc){
113                                                         if(!closure.reachable(i,j)){
114                                                                 /*
115                                                                  * the vertex j is dominated by at least one vertex outside of its SCC, 
116                                                                  * therefore it is not in the Schwartzset.
117                                                                  */
118                                                                 schwartz=false;
119                                                                 break;
120                                                         }
121                                                 }
122                                                 //StdOut.printf("%d is not in the schwarz set", i);
123                                 }
124                                 checked.add(scc);
125                                 if(schwartz){
126                                         schwartzlist.add(scc);
127                                 }
128                         }
129                 }
130                 
131                 /*
132                  * schwartzlist now contains all SCCs that are in the Schwartzset
133                  * Next we have to get all vertices contained in those SCCs
134                  */
135                 for(int i=0; i<n; i++){
136                         if(schwartzlist.contains(sccs.id(i))){
137                                 schwartzset.add(i);
138                         }
139                 }
140                        
141                 /*
142                  * Write information about the Schwartzset to the logFile.
143                  */
144                 logFile.write("Vertices in Schwartzset: "+schwartzset.size()+"\n");
145                 logFile.write("SCCs in Schwartzset: "+schwartzlist.size()+"\n");
146        
147                return schwartzset;
148         }

151        /**
152         * Writes a graph to one single txt-file in the format used as input for the MapReduce algorithms.
153         * 
154         * @param file Destination file of the graph. 
155         * @param graph The graph that is written to the file. 
156         * 
157         */
158        public static void writeGraphToFile(File file, Digraph graph){
159                
160                //used to store the adjacent vertices of one vertex.
161                List<Integer> adj = new ArrayList<Integer>();
162                
163                //number of vertices in graph
164                int n=graph.V();
165                 
166                 try {
167                        file.createNewFile();
168                        FileOutputStream stream = new FileOutputStream(file);
169                        BufferedWriter writer = new BufferedWriter(new OutputStreamWriter(stream));
170                        
171                        
172                        for (int i = 0; i < n; i++) {
173                                writer.write(i + "\t");
174                                adj.clear();
175                                for(int v : graph.adj(i)){
176                                        adj.add(v);
177                                }
178                                for(int j = 0; j<n; j++){
179                                        if (adj.contains(j)){
180                                                writer.write("1");
181                                        }else{
182                                                writer.write("0");
183                                        }
184                                        if(j!=(n-1)) writer.write(",");
185                                }
186                                writer.newLine();
187                        }
188                 
189                        writer.close();
190                 
191                } catch (IOException e) {
192                        System.out.println("some error while writing to file");
193                }
194        }

197        /**
198         * Writes a graph to several files. maxEntries is the maximum number of entries per file and 
199         * should be a multiple of the number of vertices in the graph. 
200         * 
201         * The naming of the files is only configured for up to 999 files. 
202         * 
203         * @param path Path where the files are written to. 
204         * @param graph Input graph. 
205         * @param maxEntries maximum number of entries per file.  
206         * 
207         */
208        public static void writeGraphToFiles(String path, Digraph graph, Integer maxEntries){
209                List<Integer> adj = new ArrayList<Integer>();
210                int n=graph.V();
211                int linesperfile=(int) Math.floor((double) maxEntries / (double) n);
212                int nFiles=(int) Math.ceil((double) n / (double) linesperfile);
213                File file = null;
214                int start, end;
215                BufferedWriter writer = null;
216                FileOutputStream stream;
217                Boolean check;
218                
219                System.out.println("linesperfile="+linesperfile+", nFiles="+nFiles+", total vertices="+n+"\n");
220                if(maxEntries<n) {
221                        System.out.println("maxEntries is smaller than number of vertices. Choose a larger number of maximum entries per File.\n");
222                }else if(nFiles>999) {
223                        System.out.println("Too many files. Try larger number of Entries per File");
224                }else{
225                 
226                 try {
227                         File folder = new File(path);
228                         if(folder.exists()){
229                                 File[] listOfFiles = folder.listFiles();
230                                 if(listOfFiles!=null){
231                                         for (int i = 0; i < listOfFiles.length; i++) {
232                                              if (listOfFiles[i].isFile()) {
233                                                  listOfFiles[i].delete();
234                                              } 
235                                         }
236                                 }
237                        }else{
238                                folder.createNewFile();
239                                
240                        }
241                         
242                        for(int f=1;f<=nFiles;f++){
243                                if(f<10){
244                                        file = new File(path+"part00"+f+".txt");
245                                }else if(f<100){
246                                        file = new File(path+"part0"+f+".txt");
247                                }else if(f<1000){
248                                        file = new File(path+"part"+f+".txt");
249                                }else{
250                                        System.out.println("Too many files.");
251                                }
252                                
253                                check=file.createNewFile();
254                                if(check==false){
255                                        file.delete();
256                                        file.createNewFile();
257                                }
258                                stream = new FileOutputStream(file);
259                                 
260                                writer = new BufferedWriter(new OutputStreamWriter(stream));
261                                
262                                start=(f-1)*linesperfile;
263                                end=f*linesperfile;
264                                if(end>n){
265                                        end=n;
266                                }
267                                
268                                for (int i = start; i < end; i++) {
269                                        writer.write(i + "\t");
270                                        adj.clear();
271                                        for(int v : graph.adj(i)){
272                                                adj.add(v);
273                                        }
274                                        for(int j = 0; j<n; j++){
275                                                if (adj.contains(j)){
276                                                        writer.write("1");
277                                                }else{
278                                                        writer.write("0");
279                                                }
280                                                if(j!=(n-1)) writer.write(",");
281                                        }
282                                        writer.newLine();
283                                }
284                                writer.close();
285                                
286                        }
287                
288                 
289                        
290                 
291                } catch (IOException e) {
292                        System.out.println("some error while writing to files");
293                }
294                }
295        }

433     * 
434     * @param set1
435     * @param set2
436     * @return true if the sets are identical, false otherwise.
437     */
438    public static Boolean check(List<Integer> set1, List<Integer> set2){
439        if(set1.size()!=set2.size()){
440                return false;
441        }else{
442                for(int i : set1){
443                        if(!set2.contains(i)){
444                                return false;
445                        }
446                }
447                for(int j : set2){
448                        if(!set1.contains(j)){
449                                return false;
450                        }
451                }
452        }
453        return true;
454

507

512               new FileOutputStream(output);

602         String[] parts;
603                // unzip(input, tmpfile);
604             BufferedReader reader = new BufferedReader(new FileReader(input));
605             String tmp="-";
606             
607                 StringBuilder newline=new StringBuilder();
608                 newline.append(stepID+";");
609                    try{
610                        // Read one text line at a time and display.
611                        
612                        while (true) {
613                            String line = reader.readLine();
614                           // System.out.print(line);
615                            if (line == null) break;
616                            if(line.contains("INFO org.apache.hadoop.mapreduce.Job (main): Running job:")){
617                                parts=line.split(":");
618                                newline.append(parts[4]+";");
619                                parts=line.split(",");
620                                newline.append(parts[0]+";");   
621                                    while(!line.contains("INFO org.apache.hadoop.mapreduce.Job (main):  map")){
622                                        line = reader.readLine();
623                                        if(line==null) {
624                                                reader.close();
625                                                writer.close();
626                                                return;
627                                        }
628                                    }
629                                parts=line.split(",");
630                                newline.append(parts[0]+";"); //map starting time
631                                
632                                while(!line.contains("INFO org.apache.hadoop.mapreduce.Job (main):  map 100% reduce")){
633                                        line = reader.readLine();
634                                        if(line==null) {
635                                                reader.close();
636                                                writer.close();
637                                                return;
638                                        }
639                                        if(line.contains("reduce 0%")){
640                                                parts=line.split(",");
641                                                tmp=parts[0];
642                                        }
643                                }
644                                parts=line.split(",");
645                                newline.append(parts[0]+";"); //map finishing time
646                                
647                                //reduce starting time:
648                                if(line.contains("reduce 0%")){
649                                        newline.append(parts[0]+";"); 
650                                }else{
651                                        newline.append(tmp+";"); 
652                                }
653                                
654                                while(!line.contains("INFO org.apache.hadoop.mapreduce.Job (main):  map 100% reduce 100%")){
655                                        line = reader.readLine();
656                                        if(line==null) {
657                                                reader.close();
658                                                writer.close();
659                                                return;
660                                        }
661                                }
