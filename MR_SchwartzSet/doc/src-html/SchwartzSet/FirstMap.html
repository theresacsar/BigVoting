001package SchwartzSet;
002import java.io.IOException;
003import java.util.ArrayList;
004import java.util.StringTokenizer;
005
006import org.apache.hadoop.conf.Configuration;
007import org.apache.hadoop.io.*;
008import org.apache.hadoop.mapreduce.Mapper;
009
010/**
011 * This is the Map Method for the first Mapreduce Round of the computation of the Schwartzset. The strict dominance matrix is read from the input files and 
012 * the datastructure needed for the following computation is created. 
013 * <br>The input files contain the strict dominance matrix by row (each file contains one or more rows). 
014 * <br>The first entry in each line of the inputfile is the row number and
015 * is read as InputKey of type {@link Text Text}. 
016 * <br>The InputValue (the remaining characters in the line) is the corresponding row of the adjacency matrix of the dominance graph.  
017 * <br>
018 * <br>
019 * The (key,value)-pairs created by this method have keys of type {@link IntWritable} and values of type {@link VertexArrayWritable}.
020 * 
021 * 
022 * @author ------
023 *
024 */
025public class FirstMap extends Mapper<Text, Text, IntWritable, VertexArrayWritable>
026        {
027
028                public void map (Text key, Text row, Context context)
029                        throws IOException, InterruptedException 
030                {
031                        
032                        StringTokenizer elements = new StringTokenizer(row.toString(),",");
033                        Configuration conf = context.getConfiguration();
034                        int N = conf.getInt("SchwartzSet.N", 1);
035                        int i;
036                        
037                        //Convert the key from Text to IntWritable
038                        IntWritable k = new IntWritable(Integer.parseInt(key.toString())); 
039
040                        //Read the row of of the adjacency matrix to the buffer
041                        ArrayList<IntWritable> buffer = new ArrayList<IntWritable>(N);
042                        for(int j=0; j<N;j++){
043                                // if the rows contain a 1
044                                if(Integer.parseInt(elements.nextToken().toString())==1){
045                                        buffer.add(new IntWritable(j));
046                                }
047                        }
048                        
049                        IntWritable[] vertices = new IntWritable[buffer.size()];
050                        IntWritable[] me = new IntWritable[1];
051                        me[0]=k;
052                        
053                        i=0;
054                        for(IntWritable element : buffer){
055                                vertices[i]=element;
056                                
057                                // tell "element", that it is reached by me
058                                context.write(element, new VertexArrayWritable(me, new IntWritable(3))); 
059                                i++;
060                        }
061                        
062                        
063                        //emit all vertices that can be reached from k as "new" to the corresponding reducer
064                        context.write(k, new VertexArrayWritable(vertices, new IntWritable(1))); 
065                }   
066        }
