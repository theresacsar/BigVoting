001package SchwartzSet;
002import java.io.IOException;
003import org.apache.hadoop.io.*;
004import org.apache.hadoop.mapreduce.Mapper;
005
006/**
007 * Last Map Method of the Schwartzset Computation. This method emits the vertex as outputvalue ({@link IntWritable})
008 * if it is contained in the Schwartzset.
009 * 
010 * @author -----
011 *
012 */
013public class LastMap extends Mapper<IntWritable, VertexWritable, NullWritable, IntWritable>
014{
015        public void map (IntWritable key, VertexWritable vertex, Context context)
016                throws IOException, InterruptedException 
017        {
018                
019                
020                IntWritable k = new IntWritable(Integer.parseInt(key.toString()));
021                
022                //IntWritable tmp;
023                Boolean Schwartzset=true, found=false;
024        
025
026                if(vertex.reachedBy.values.length<=vertex.old.values.length+vertex.neu.values.length){
027
028                        for(IntWritable i : vertex.reachedBy.values){
029                                found=false;
030                                if(i.equals(k)) {
031                                        found=true;
032                                }else{
033                                        for(int j=0; j<vertex.neu.values.length; j++){
034                                                if(vertex.neu.values.equals(i)){
035                                                        found = true;
036                                                        break;
037                                                }
038                                        }
039                                }
040                                if(!found){
041                                        for(int j=0; j<vertex.old.values.length; j++){
042                                                if(vertex.old.values.equals(i)){
043                                                        found = true;
044                                                        break;
045                                                }
046                                        }
047                                }
048                                if(!found) {
049                                        Schwartzset=false;
050                                        break;
051                                }
052                        }
053                }else{
054                        Schwartzset=false;
055                }
056                
057                if(Schwartzset){
058                        context.write(NullWritable.get(), k);
059                }
060        }   
061}
