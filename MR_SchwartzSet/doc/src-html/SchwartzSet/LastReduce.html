001package SchwartzSet;
002import java.io.IOException;
003import org.apache.hadoop.io.IntWritable;
004import org.apache.hadoop.io.NullWritable;
005import org.apache.hadoop.mapreduce.Reducer;
006
007/**
008 * This last reduce method collects all vertices contained in the Schwartzset.
009 * 
010 * @author --------
011 *
012 */
013public class LastReduce extends Reducer<NullWritable, IntWritable, NullWritable, IntWritable> {
014        
015        protected void reduce(NullWritable key, java.lang.Iterable<IntWritable> vertices, org.apache.hadoop.mapreduce.Reducer<NullWritable, IntWritable, NullWritable, IntWritable>.Context context)
016        
017        throws IOException, InterruptedException {
018
019                for (IntWritable v: vertices){
020                                context.write(NullWritable.get(), v);
021                        }
022                }
023        }
